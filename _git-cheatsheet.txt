<<< OPERATIONS >>>
git init
git add *
git add <file>
git commit -m 'this is initial commit'
git status
git status -s

git restore --staged <file>
git reset HEAD <file>
	both of them unstage a file

git checkout -- <file>
git restore <file>
	both of them drop changes

git commit --amend


rm <file>
git rm <file>
	if I want to del a staged file, this will del file from $PWD and staged
git rm <file> -f
	does it in one step

git mv <oldFile> <newFile>
	actually does mv, git rm, git add


<<< VIEWS >>>
git diff
	between $PWD and staged
git diff --staged
	between staged and HEAD
git commit -a -m 'small changes'
	adds all tracked files and then commits


git log -p -1
git log --stat -1
	see log of commits, limited to one, including diffs || stats
git log --pretty=oneline
	one commit per line :)
